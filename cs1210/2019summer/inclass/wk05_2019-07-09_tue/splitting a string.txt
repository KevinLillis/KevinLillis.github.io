# Splitting a string

river = 'mississippi river'

# Default splits on any number of white space characters
river.split()
['mississippi', 'river']

river = 'mississippi       river'

river.split()
['mississippi', 'river']

river = 'mississippi\nriver'

river.split()
['mississippi', 'river']

river = 'mississippi\n\t\t   \n  \triver'
print(river)
mississippi
		   
  	river

river.split()
['mississippi', 'river']

river = 'mississippi river'

# Can split on other substrings as well
river.split('ss')
['mi', 'i', 'ippi river']

# Using split with multiple assignment
