void match(Token t) {
    // moves lookAhead pointer
    if (lookAhead == t)
        lookAhead = nextToken();
    else error();
}

void decl() {
    type();
    var-list();
}

void type() {
    if (lookAhead == int)
        match(int);
    else if (lookAhead == double)
        match(double);
    else error();
}

void var-list() {
    id();
    if (lookAhead == ,) {
        match(,);
        var-list();
    }
}

void id(){
    if (lookAhead == a)
        match(a);
    else if (lookAhead == b)
        match(b);
    else if (lookAhead == c)
        match(c);
    else error();
}

void error() {
    exit(EXIT_FAILURE);
}
