Pseudo-Code for Recursive Descent Parser

Token lookAhead;

.
.
.

void match(Token t) {
    // moves lookAhead pointer
    if (lookAhead == t)
        lookAhead = nextToken();
    else error();
}

void decl() {
    type();
    varlist();
}

void type() {
    if (lookAhead == int)
        match(int);
    else if (lookAhead == double)
        match(double);
    else error();
}

void varlist() {
    id();
    if (lookAhead == ,) {
        match(,);
        varlist();
    }
}

void id(){
    if (lookAhead == a)
        match(a);
    else if (lookAhead == b)
        match(b);
    else if (lookAhead == c)
        match(c);
    else error();
}

void error() {
    exit(EXIT_FAILURE);
}
